  • # List comprehensions
>>> fruits = ['Banana', 'Apple', 'Strawberry']
>>> fruits = [fruit.strip() for fruit in fruits]
>>> print(fruits)
['Banana', 'Apple', 'Strawberry']

# List and the enumerate function
>>> list(enumerate(fruits))
[(0, 'Banana'), (1, 'Apple'), (2, 'Strawberry')]
